\chapter{x86-64 速查表}

\section{通用寄存器}

\begin{center}
\begin{tabular}{ll}
\toprule
寄存器 & 常见用途 \\
\midrule
\code{rax/eax} & 返回值、通用计算 \\
\code{rdi/edi} & System V 第 1 个整数/指针参数 \\
\code{rsi/esi} & System V 第 2 个整数/指针参数 \\
\code{rdx/edx} & System V 第 3 个整数/指针参数 \\
\code{rcx/ecx} & System V 第 4 个整数/指针参数 \\
\code{r8/r8d} & System V 第 5 个整数/指针参数 \\
\code{r9/r9d} & System V 第 6 个整数/指针参数 \\
\code{rsp} & 栈指针 \\
\code{rbp} & 栈帧指针或通用 callee-saved 寄存器 \\
\bottomrule
\end{tabular}
\end{center}

\section{常见指令}

\begin{longtable}{lll}
\toprule
指令 & 简化语义 & 备注 \\
\midrule
\code{mov dst, src} & \code{dst = src} & 复制位模式 \\
\code{lea dst, [addr]} & \code{dst = addr} & 不访问内存 \\
\code{add dst, src} & \code{dst += src} & 更新 flags \\
\code{sub dst, src} & \code{dst -= src} & 更新 flags \\
\code{cmp a, b} & 计算 \code{a-b} 更新 flags & 不保存结果 \\
\code{test a, b} & 计算 \code{a\&b} 更新 flags & 常用于判零 \\
\code{jmp label} & 无条件跳转 & 改变 \code{rip} \\
\code{call f} & 调用函数 & 压入返回地址 \\
\code{ret} & 返回 & 弹出返回地址 \\
\bottomrule
\end{longtable}
