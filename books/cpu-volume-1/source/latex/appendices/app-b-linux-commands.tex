\chapter{Linux 与二进制工具命令速查}

\section{编译与反汇编}

\begin{lstlisting}[language=bash]
g++ -E main.cpp -o main.i
g++ -S -O3 -masm=intel main.cpp -o main.s
g++ -c -O3 main.cpp -o main.o
g++ main.o -o app
objdump -drwC -Mintel main.o
\end{lstlisting}

\section{ELF 和符号}

\begin{lstlisting}[language=bash]
file app
readelf -h app
readelf -S app
readelf -l app
readelf -r app
nm -C app
\end{lstlisting}

\section{调试}

\begin{lstlisting}[language=bash]
g++ -O0 -g main.cpp -o app
gdb -q ./app
\end{lstlisting}

\begin{lstlisting}
break main
run
stepi
info registers
x/10i $rip
disassemble /m main
\end{lstlisting}
